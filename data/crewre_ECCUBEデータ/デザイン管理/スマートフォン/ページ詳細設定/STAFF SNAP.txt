<script>
	$(function(){
		let now;
		$(document).on('click','.snap_btn',function(){
			let pid = $(this).data('snap');
			now = $(this).data('snapindex');
			next = now + 1;
			prev = now - 1;
			$('#snap_iframe').attr('src', '/contents/coordinate_detail.php?cid='+pid);
			$('#snap_block_outer').fadeIn('fast');
			if(typeof $('.snapindex' + next).data('snap') === 'undefined'){
				$('.btn_snap_r').hide();
			}
			if(typeof $('.snapindex' + prev).data('snap') === 'undefined'){
				$('.btn_snap_l').hide();
			}
		});
		$(document).on('click','.btn_snap_r', function(){
			now = now + 1;
			let next = now + 1;
			$('#snap_iframe').attr('src', '');
			pid = $('.snapindex' + now).data('snap');
			$('#snap_iframe').attr('src', '/contents/coordinate_detail.php?cid='+pid);
			if(typeof $('.snapindex' + next).data('snap') === 'undefined'){
				$('.btn_snap_r').hide();
			}else{
				$('.btn_snap_r').show();
			}
			$('.btn_snap_l').show();
		});
		$(document).on('click','.btn_snap_l', function(){
			now = now - 1;
			let prev = now - 1;
			$('#snap_iframe').attr('src', '');
			pid = $('.snapindex' + now).data('snap');
			$('#snap_iframe').attr('src', '/contents/coordinate_detail.php?cid='+pid);
			if(typeof $('.snapindex' + prev).data('snap') === 'undefined'){
				$('.btn_snap_l').hide();
			}else{
				$('.btn_snap_l').show();
			}
			$('.btn_snap_r').show();
		});
		$(document).on('click','.btn_outer_close, #snap_block_bg',function(){
			$('#snap_block_outer').fadeOut('fast',function(){
				$('#snap_iframe').attr('src', '');
			});
		});
	});
</script>
